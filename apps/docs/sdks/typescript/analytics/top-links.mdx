<AnalyticsArguments />

## Response

<ResponseField name="link" type="string" required="true">
  The unique ID of the short link
</ResponseField>

<ResponseField name="clicks" type="number" required="true">
  The number of clicks from this link
</ResponseField>

<RequestExample>

```typescript
await dub.analytics.topLinks({
  key: "dummy",
  domain: "dub.sh",
  interval: "30d",
});
```

</RequestExample>

<ResponseExample>

```typescript
[
  { link: "cluvb4cwi0001sd593yfdl1sz", clicks: 19 },
  { link: "clri4mgix0001l608ru12tcfr", clicks: 8 },
];
```

</ResponseExample>
